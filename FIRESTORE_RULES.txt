// ===========================================================
// BaatBanao — Firestore Security Rules (v1.0.9 chat system)
// Paste this into: Firebase Console > Firestore Database > Rules > Publish
// ===========================================================

rules_version = '2';
service cloud.firestore {
  match /databases/{database}/documents {

    function isSignedIn() {
      return request.auth != null;
    }

    // ---------- users/{uid} ----------
    // Stores: uid, displayName, connectCode, createdAt, lastSeen
    // Read is open to any signed-in user because the app looks up
    // a partner by their 6-char connectCode (client-side query,
    // no backend function). Only the owner can create/update their doc.
    match /users/{userId} {
      allow read: if isSignedIn();
      allow create: if isSignedIn() && request.auth.uid == userId;
      allow update: if isSignedIn() && request.auth.uid == userId;
      allow delete: if false;
    }

    // ---------- chats/{chatId} ----------
    // chatId = sorted "uidA_uidB". Only the two participants can
    // read/write their own chat doc (metadata: lastMessage, unreadCount, etc.)
    //
    // IMPORTANT: `resource` is null when the document doesn't exist yet.
    // The very first thing the app does when connecting two users is
    // `getDoc(chatRef)` to check "does this chat already exist?" — BEFORE
    // it has been created. Without the `resource == null` short-circuit
    // below, evaluating `resource.data.participants` on a non-existent
    // doc throws inside the rules engine and Firestore reports that as
    // "permission-denied" instead of "not found", which permanently
    // breaks starting any brand-new chat.
    match /chats/{chatId} {
      allow read: if isSignedIn() &&
        (resource == null || request.auth.uid in resource.data.participants);
      allow create: if isSignedIn() && request.auth.uid in request.resource.data.participants;
      allow update: if isSignedIn() && request.auth.uid in resource.data.participants;
      allow delete: if false;

      // ---------- chats/{chatId}/messages/{messageId} ----------
      match /messages/{messageId} {
        allow read: if isSignedIn() &&
          request.auth.uid in get(/databases/$(database)/documents/chats/$(chatId)).data.participants;

        allow create: if isSignedIn() &&
          request.auth.uid in get(/databases/$(database)/documents/chats/$(chatId)).data.participants &&
          request.resource.data.senderId == request.auth.uid;

        // update needed for "mark as read" (readBy array + status)
        allow update: if isSignedIn() &&
          request.auth.uid in get(/databases/$(database)/documents/chats/$(chatId)).data.participants;

        allow delete: if false;
      }
    }
  }
}
